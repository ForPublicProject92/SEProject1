lib/
├─ main.dart             # 앱 진입점
├─ pages/                 # 화면 단위 페이지
│   ├─ start.dart 			 # 앱 시작화면 /  자동 로그인으로 선택되어 있으면 바로 main_page.dart로 스킵
│   ├─ select.dart    			 # 로그인/ 회원가입 선택 화면
│   ├─ login.dart        		 # 로그인 화면  
│   ├─ signup.dart                  # 회원가입 화면 
│   ├─ main_page.dart		 # 메인 화면 / 필요한 기능 : 동적 배경,오브젝트 , 텍스트, 텍스트 입력 및 출력 기능 
│   ├─ configuration.dart 	 # 설정 화면 / 알람이나 개인 정보를 설정하고 가족 코드를 통해 가족을 연결
│   ├─ calendar.dart 		 # 달력 화면  / 달력을 클릭시 그날 받은 질문과 주고 받은 메세지를 확인 가능
│   └─ record.dart 			 # 메세지 기록 화면 / 그동안 주고 받은 질문과 메세지를 리스트로 정렬
├─ widgets/            # 재사용 가능한 작은 UI 위젯
│   ├─ button.dart
│   └─ card.dart
├─ models/             # 데이터 모델 클래스
├─ services/           # 백엔드 API 호출, 데이터 처리
└─ utils/                 # 공통 함수나 상수
